  <body>

    <h3>Plastic Detection</h3>

    <video id="webcam" autoplay muted playsinline></video>

    <br />

    <button onclick="predict()">Predict</button>

    <p id="result">Waiting...</p>


    <script>

      const URL = "https://teachablemachine.withgoogle.com/models/T7C-BMBxH/"; 

      let model, webcam;


      async function init() {

        const modelURL = URL + "model.json";

        const metadataURL = URL + "metadata.json";


        model = await tmImage.load(modelURL, metadataURL);

        webcam = new tmImage.Webcam(200, 200, true);

        await webcam.setup();

        await webcam.play();

        document.getElementById("webcam").srcObject = webcam.webcam;

        window.requestAnimationFrame(loop);

      }


      async function loop() {

        webcam.update();

        window.requestAnimationFrame(loop);

      }


      async function predict() {

        const prediction = await model.predict(webcam.canvas);

        const top = prediction.reduce((prev, current) =>

          (prev.probability > current.probability) ? prev : current);

        document.getElementById("result").innerText =

          `${top.className} (${(top.probability * 100).toFixed(2)}%)`;


        // Send result to App Inventor

        if (window.AppInventor) {

          window.AppInventor.setWebViewString(top.className);

        }

      }


      init();

    </script>

  </body>
